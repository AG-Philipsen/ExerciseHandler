%Give an error when a language is used for which no translation exists
\newcommand{\CheckTranslation}[1]{%
  \IfTranslation{\languagename}{#1}{}{%
    \PackageError{Vocabulary}{%
      No available translation for\MessageBreak
      \space\space\space "#1"\MessageBreak
      in the used language (\languagename).\MessageBreak
      Vocabulary is not really a package.\MessageBreak
      This error means that you are using\MessageBreak
      a language that is not supported in\MessageBreak
      the present version of the Exercise\MessageBreak
      Handler. See README for more help!
    }{%
      More information can be found in the README file.
    }
  }%
}

%Capitalize first letter (inspired from https://tex.stackexchange.com/a/380505
\newcommand\firstUp[1]{\expandafter\firstUpAux#1\relax}
\def\firstUpAux#1#2\relax{\uppercase{#1}#2}

%Wrapper for \GetTranslation in order to have capital letter variant
\newcommand{\translate}[1]{\CheckTranslation{#1}\GetTranslation{#1}}
\newcommand{\Translate}[1]{\CheckTranslation{#1}\SaveTranslation{\tmpVar}{#1}\firstUp{\tmpVar}}
